```python
import sys
from collections import deque

input=sys.stdin.readline
N = int(input())
q = deque([])

for i in range(N) :
    order = input().strip()
    l = len(q)

    if 'push_front' in order :
        q.appendleft(int(order.split()[1]))
    elif 'push_back' in order :
        q.append(int(order.split()[1]))
    elif 'back' == order :
        if l == 0 :
            print("-1")
        else:
            print(q[-1])
    elif 'front' == order :
        if l == 0 :
            print("-1")
        else:
            print(q[0])
    elif 'size' == order:
        print(l)
    elif 'empty' == order:
        if l == 0 :
            print("1")
        else :
            print("0")
    elif 'pop_front' == order:
        if l ==0:
            print("-1")
        else:
            print(q.popleft())
    elif 'pop_back' == order:
        if l ==0:
            print("-1")
        else:
            print(q.pop())
```
