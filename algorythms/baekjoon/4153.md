```python
import sys, math

input = sys.stdin.readline

  
  

while True :

    ipt = input().strip()

    if ipt == "0 0 0":

        break

    else:

        arr = list(map(int,ipt.split()))

        arr.sort()

        if math.pow(arr[0],2) + math.pow(arr[1],2) == math.pow(arr[2],2) :

            print("right")

        else:

            print("wrong")
```
