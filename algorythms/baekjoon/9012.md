```python
N = int(input())

for i in range(N) :
    string = input()
    cnt = 0
    limit = len(string)//2+1

    while '()' in string or cnt < limit:
        string = string.replace("()","")
        cnt += 1

    if string == '()' or string == "" :
        print("YES")
    else:
        print("NO")
```
